<script>
const GS={START:'start',FIELD:'field',BASE:'base',DEAD:'dead',RAID:'raid'};
const CELL=30,ROWS=20,FCOLS=14;
const BASE_W=200,FW=FCOLS*CELL,CH=ROWS*CELL,CW=BASE_W+FW+BASE_W;
const DOOR=[8,9,10,11];
const ROUND_DURATION=3*60*1000;

let canvas,ctx,last=-1;
let gs=GS.START;

let P=[],pDir={x:1,y:0},pNd={x:1,y:0},pStep=0,pSpeed=195;
let pRes=0,pCarry=0,dockedLen=3;
let upg={lives:0,speed:0,armor:0};

let roundTimer=ROUND_DURATION;
let roundNum=1;
let pBaseStr=0;
let aiBaseStr=0;
let raidResult=null;

const UDEFS=[
  {name:'Подкрепление',desc:'+1 HP каждому бойцу',costs:[50,100,150],max:3,k:'lives'},
  {name:'Движок',desc:'Скорость колонны +25%',costs:[75,150],max:2,k:'speed'},
  {name:'Броня',desc:'Блок 50% входящего урона',costs:[100],max:1,k:'armor'},
];
let baseTab=0;

let AI=[],aiDir={x:-1,y:0},aiStep=0;
let aiMode='collect',aiModeT=0,aiTarget=null;
let aiRes=0;

let items=[],haz=[],expl=[],fx_list=[];
let itemT=0,hazT=0;

// анимация бойцов в базе
let baseAnimT=0;
let baseSoldierAnim=[];

window.onload=()=>{
  canvas=document.getElementById('c');
  canvas.width=CW;canvas.height=CH;
  ctx=canvas.getContext('2d');
  document.addEventListener('keydown',onKey);
  requestAnimationFrame(loop);

const ri=(a,b)=>Math.floor(Math.random()*(b-a+1))+a;

function initGame(){
  const mhp=()=>3+upg.lives;
  // Бойцы начинают В БАЗЕ — координаты внутри базовой панели (виртуальные, не на поле)
  P=[{x:2,y:9,hp:mhp()},{x:1,y:9,hp:mhp()},{x:0,y:9,hp:mhp()}];
  pDir={x:1,y:0};pNd={x:1,y:0};pStep=0;pSpeed=195;
  pRes=0;pCarry=0;dockedLen=3;
  upg={lives:0,speed:0,armor:0};baseTab=0;
  pBaseStr=0;aiBaseStr=0;aiRes=0;
  roundTimer=ROUND_DURATION;roundNum=1;raidResult=null;

  AI=[{x:FCOLS-3,y:9},{x:FCOLS-2,y:9},{x:FCOLS-1,y:9}];
  aiDir={x:-1,y:0};aiStep=0;
  aiMode='collect';aiModeT=ri(9000,14000);aiTarget=null;

  items=[];haz=[];expl=[];fx_list=[];
  itemT=ri(1200,2000);hazT=ri(8000,12000);
  for(let i=0;i<8;i++)spawnItem();

  initBaseSoldiers();
  // Игра начинается с базы — бойцы ещё не вышли
  gs=GS.BASE;
}

function initBaseSoldiers(){
  baseSoldierAnim=[];
  for(let i=0;i<3;i++){
    baseSoldierAnim.push({
      bx:20+i*48,by:200,
      phase:Math.random()*Math.PI*2,
      state:'idle' // idle, march
    });
  }
}

function nextRound(){
  const mhp=3+upg.lives;
  dockedLen=Math.max(3,P.length);
  pCarry=0;
  pDir={x:1,y:0};pNd={x:1,y:0};pStep=0;
  P=[];
  for(let i=0;i<dockedLen;i++){
    const col=i%3,row=Math.floor(i/3);
    P.push({x:2-col,y:9+row,hp:mhp});
  }
  AI=[{x:FCOLS-3,y:9},{x:FCOLS-2,y:9},{x:FCOLS-1,y:9}];
  aiMode='collect';aiModeT=ri(9000,14000);aiTarget=null;
  aiStep=0;
  items=[];haz=[];expl=[];fx_list=[];
  itemT=ri(1200,2000);
  hazT=Math.max(5000,ri(8000,12000)-roundNum*500);
  for(let i=0;i<8;i++)spawnItem();
  roundTimer=ROUND_DURATION;
  roundNum++;
  initBaseSoldiers();
  gs=GS.BASE;
}

function onKey(e){
  if(e.code==='KeyR'){initGame();return;}
  if(gs===GS.START||gs===GS.DEAD){
    if(e.code==='Space'||e.code==='Enter')initGame();
    return;
  }
  if(gs===GS.RAID){
    if(e.code==='Space'||e.code==='Enter')nextRound();
    return;
  }
  if(gs===GS.BASE){
    if(e.code==='ArrowUp'||e.code==='KeyW')baseTab=Math.max(0,baseTab-1);
    if(e.code==='ArrowDown'||e.code==='KeyS')baseTab=Math.min(2,baseTab+1);
    if(e.code==='Space'||e.code==='Enter')buyUpg(baseTab);
    if(e.code==='KeyE'||e.code==='Escape')exitBase();
    e.preventDefault();return;
  }
  if(gs!==GS.FIELD)return;
  const M={
    ArrowUp:{x:0,y:-1},KeyW:{x:0,y:-1},
    ArrowDown:{x:0,y:1},KeyS:{x:0,y:1},
    ArrowLeft:{x:-1,y:0},KeyA:{x:-1,y:0},
    ArrowRight:{x:1,y:0},KeyD:{x:1,y:0}
  };
  if(M[e.code]){
    const m=M[e.code];
    if(!(m.x===-pDir.x&&pDir.x!==0)&&!(m.y===-pDir.y&&pDir.y!==0))pNd=m;
  }
  if(['Space','ArrowUp','ArrowDown','ArrowLeft','ArrowRight'].includes(e.code))e.preventDefault();
}

function upgCost(i){const d=UDEFS[i],l=upg[d.k];return l>=d.max?null:d.costs[l];}
function buyUpg(i){
  const cost=upgCost(i);if(cost===null||pRes<cost)return;
  pRes-=cost;upg[UDEFS[i].k]++;
  if(UDEFS[i].k==='speed')pSpeed=Math.max(105,195-upg.speed*30);
  pBaseStr+=cost*0.5;
  fxAdd(BASE_W/2,280,UDEFS[i].name+' ✓','#FFD700',1600);
}

function exitBase(){
  const mhp=3+upg.lives;
  const len=Math.max(2,dockedLen);
  P=[];
  for(let i=0;i<len;i++){
    const col=i%3,row=Math.floor(i/3);
    P.push({x:2-col,y:9+row,hp:mhp});
  }
  pDir={x:1,y:0};pNd={x:1,y:0};pCarry=0;
  // Анимация: бойцы маршируют на выход
  baseSoldierAnim.forEach(s=>s.state='march');
  gs=GS.FIELD;
}

function occ(){
  const s=new Set();
  P.forEach(p=>s.add(p.x+','+p.y));
  AI.forEach(p=>s.add(p.x+','+p.y));
  items.forEach(p=>s.add(p.x+','+p.y));
  haz.forEach(h=>s.add(h.x+','+h.y));
  return s;
}
function emptyCell(){
  const o=occ();
  for(let t=0;t<300;t++){
    const x=ri(1,FCOLS-2),y=ri(0,ROWS-1);
    if(!o.has(x+','+y))return{x,y};
  }
  return null;
}
const ITYPES=[
  {t:'supply',w:5,v:12},{t:'crate',w:3,v:35},
  {t:'medkit',w:2,v:0},{t:'recruit',w:2,v:20},
];
function spawnItem(){
  const c=emptyCell();if(!c)return;
  const tot=ITYPES.reduce((s,t)=>s+t.w,0);
  let r=Math.random()*tot,ch=ITYPES[0];
  for(const t of ITYPES){r-=t.w;if(r<=0){ch=t;break;}}
  items.push({...c,t:ch.t,v:ch.v});
}

function spawnHaz(){
  const count=roundNum>=2?2:1;
  for(let i=0;i<count;i++){
    const c=emptyCell();if(!c)return;
    haz.push({x:c.x,y:c.y,w:2400,mw:2400,done:false});
  }
}
function explodeHaz(h){
  h.done=true;
  expl.push({x:h.x,y:h.y,l:700,ml:700});
  for(let i=P.length-1;i>=0;i--){
    if(Math.abs(P[i].x-h.x)<=1&&Math.abs(P[i].y-h.y)<=1){damage(1);break;}
  }
  for(let i=AI.length-1;i>=0;i--){
    if(Math.abs(AI[i].x-h.x)<=1&&Math.abs(AI[i].y-h.y)<=1){
      if(AI.length>1)AI.pop();
      fxAdd(cx(h.x),cy(h.y),'-AI боец','#FF8888',700);break;
    }
  }
}

function damage(n){
  if(upg.armor&&Math.random()<0.5){
    if(P.length)fxAdd(cx(P[0].x),cy(P[0].y),'БРОНЯ!','#8888FF',700);
    return;
  }
  for(let k=0;k<n;k++){
    if(!P.length){gs=GS.DEAD;return;}
    const tail=P[P.length-1];
    if(tail.hp>1){
      tail.hp--;
      fxAdd(cx(tail.x),cy(tail.y),'-HP','#FF5555',700);
    } else {
      P.pop();
      if(!P.length){gs=GS.DEAD;return;}
      fxAdd(cx(P[0].x),cy(P[0].y),'-1 боец!','#FF3333',1000);
    }
  }
}

function calcBaseStr(){
  const mhp=3+upg.lives;
  let str=pRes;
  P.forEach(s=>str+=s.hp*8);
  str+=upg.lives*30+upg.speed*25+upg.armor*40;
  pBaseStr=Math.round(str);
}
function calcAiBaseStr(){
  let str=aiRes;
  AI.forEach(()=>str+=24);
  aiBaseStr=Math.round(str);
}

function movePlayer(){
  pDir={...pNd};
  const h=P[0];
  let nx=h.x+pDir.x,ny=h.y+pDir.y;
  if(nx<0&&DOOR.includes(ny)){enterBase();return;}
  if(nx<0||nx>=FCOLS||ny<0||ny>=ROWS){damage(1);return;}
  if(gs===GS.DEAD)return;
  for(let i=0;i<P.length-1;i++){
    if(P[i].x===nx&&P[i].y===ny){damage(2);return;}
  }
  if(gs===GS.DEAD)return;
  if(AI.length>0&&AI[0].x===nx&&AI[0].y===ny){
    damage(1);
    if(AI.length>1){
      AI.pop();
      fxAdd(cx(nx),cy(ny),'ТАРАН!','#FFAA00',1000);
    } else {
      AI=[{x:FCOLS-3,y:9},{x:FCOLS-2,y:9},{x:FCOLS-1,y:9}];
      aiMode='collect';aiModeT=ri(9000,14000);aiTarget=null;
      fxAdd(cx(nx),cy(ny),'AI УНИЧТОЖЕН!','#FF4400',1500);
    }
    return;
  }
  if(AI.some(s=>s.x===nx&&s.y===ny)){damage(1);return;}
  if(gs===GS.DEAD)return;
  P.unshift({x:nx,y:ny,hp:h.hp});
  let grow=false;
  const ii=items.findIndex(i=>i.x===nx&&i.y===ny);
  if(ii!==-1){const it=items.splice(ii,1)[0];grow=collect(it);}
  if(!grow)P.pop();
}

function collect(it){
  if(it.t==='supply'){pCarry+=it.v;fxAdd(cx(it.x),cy(it.y),'+'+it.v,'#88FF88',900);return false;}
  if(it.t==='crate'){pCarry+=it.v;fxAdd(cx(it.x),cy(it.y),'+'+it.v,'#FFD700',900);return false;}
  if(it.t==='medkit'){
    const mhp=3+upg.lives;
    for(let i=P.length-1;i>=0;i--){if(P[i].hp<mhp){P[i].hp=Math.min(mhp,P[i].hp+1);break;}}
    fxAdd(cx(it.x),cy(it.y),'❤+1','#FF7777',900);return false;
  }
  if(it.t==='recruit'){
    pCarry+=it.v;
    P.push({x:P[P.length-1].x,y:P[P.length-1].y,hp:3+upg.lives});
    fxAdd(cx(it.x),cy(it.y),'+боец','#88FF44',1000);return true;
  }
  return false;
}

function enterBase(){
  dockedLen=Math.max(2,P.length);
  pRes+=pCarry;
  if(pCarry>0)fxAdd(BASE_W/2,270,'+'+pCarry+' рес.','#FFD700',2000);
  pCarry=0;gs=GS.BASE;baseTab=0;
  initBaseSoldiers();
}

function updateAI(dt){
  aiModeT-=dt;
  if(aiModeT<=0){
    aiMode=aiMode==='collect'?'return':'collect';
    aiModeT=aiMode==='collect'?ri(9000,14000):ri(3000,5000);
    aiTarget=null;
  }
  aiStep+=dt;if(aiStep<255)return;aiStep=0;
  const h=AI[0];
  let tx,ty;
  if(aiMode==='return'){
    tx=FCOLS-1;ty=9;
    if(h.x>=FCOLS-1){
      AI=[{x:FCOLS-3,y:9},{x:FCOLS-2,y:9},{x:FCOLS-1,y:9}];
      aiMode='collect';aiModeT=ri(8000,13000);aiTarget=null;
      return;
    }
  } else {
    if(!aiTarget||!items.find(i=>i.x===aiTarget.x&&i.y===aiTarget.y)){
      if(items.length&&Math.random()<0.70){
        let best=null,bd=999;
        items.forEach(i=>{const d=Math.abs(i.x-h.x)+Math.abs(i.y-h.y);if(d<bd){bd=d;best=i;}});
        aiTarget=best;
      } else {
        aiTarget={x:ri(2,FCOLS-3),y:ri(1,ROWS-2)};
      }
    }
    tx=aiTarget?.x??FCOLS/2;ty=aiTarget?.y??ROWS/2;
  }
  const dx=tx-h.x,dy=ty-h.y;
  let nd;
  if(Math.random()<0.22){
    const opts=[{x:1,y:0},{x:-1,y:0},{x:0,y:1},{x:0,y:-1}];
    nd=opts[ri(0,3)];
  } else {
    nd=Math.abs(dx)>Math.abs(dy)?{x:Math.sign(dx),y:0}:{x:0,y:Math.sign(dy)};
  }
  let nx=h.x+nd.x,ny=h.y+nd.y;
  if(AI.some((s,i)=>i<AI.length-1&&s.x===nx&&s.y===ny)||nx<0||nx>=FCOLS||ny<0||ny>=ROWS){
    nd={x:-nd.y,y:nd.x};nx=h.x+nd.x;ny=h.y+nd.y;
  }
  nx=Math.max(0,Math.min(FCOLS-1,nx));
  ny=Math.max(0,Math.min(ROWS-1,ny));
  AI.unshift({x:nx,y:ny});
  const ii=items.findIndex(i=>i.x===nx&&i.y===ny);
  if(ii!==-1){aiRes+=items[ii].v;items.splice(ii,1);}
  AI.pop();
}

function triggerRaid(){
  calcBaseStr();
  calcAiBaseStr();
  const win=pBaseStr>=aiBaseStr;
  raidResult={pStr:pBaseStr,aiStr:aiBaseStr,win,round:roundNum};
  gs=GS.RAID;
}

function fxAdd(x,y,txt,col,life){fx_list.push({x,y,txt,col,life,ml:life});}
function cx(x){return BASE_W+x*CELL+CELL/2;}
function cy(y){return y*CELL+CELL/2;}

function update(dt){
  fx_list=fx_list.filter(f=>{f.life-=dt;return f.life>0;});
  baseAnimT+=dt;

  if(gs===GS.DEAD||gs===GS.START||gs===GS.RAID)return;

  if(gs===GS.BASE){
    // анимируем бойцов в базе
    baseSoldierAnim.forEach(s=>{
      s.phase+=dt*0.003;
      if(s.state==='march')s.bx=Math.min(BASE_W-20,s.bx+dt*0.06);
    });
    return;
  }

  roundTimer-=dt;
  if(roundTimer<=0){triggerRaid();return;}

  itemT-=dt;if(itemT<=0&&items.length<12){spawnItem();itemT=ri(2000,3500);}
  hazT-=dt;if(hazT<=0){spawnHaz();hazT=Math.max(5000,ri(8000,12000)-roundNum*400);}
  haz.forEach(h=>{if(!h.done){h.w-=dt;if(h.w<=0)explodeHaz(h);}});
  haz=haz.filter(h=>!h.done);
  expl=expl.filter(e=>{e.l-=dt;return e.l>0;});
  pStep+=dt;if(pStep>=pSpeed){pStep-=pSpeed;movePlayer();}
  updateAI(dt);
}

function drawBaseSoldier(bx,by,phase,isHead){
  const bobY=Math.sin(phase)*2;
  ctx.fillStyle='#2A4F14';
  ctx.fillRect(bx-10,by+bobY-8,20,18);
  // голова
  ctx.fillStyle='#3E6E1A';
  ctx.beginPath();ctx.ellipse(bx,by+bobY-14,8,6,0,Math.PI,0);ctx.fill();
  ctx.fillRect(bx-10,by+bobY-14,20,4);
  // глаза
  ctx.fillStyle='#88D040';
  ctx.fillRect(bx-5,by+bobY-11,3,3);
  ctx.fillRect(bx+2,by+bobY-11,3,3);
  // ноги — шагающая анимация
  const legOff=Math.sin(phase*2)*4;
  ctx.fillStyle='#1A3A0A';
  ctx.fillRect(bx-7,by+bobY+10,5,8+legOff);
  ctx.fillRect(bx+2,by+bobY+10,5,8-legOff);
  // оружие
  ctx.fillStyle='#222';
  ctx.fillRect(bx+8,by+bobY-8,3,14);
  ctx.fillStyle='#111';
  ctx.fillRect(bx+5,by+bobY-10,6,4);
  if(isHead){
    ctx.strokeStyle='rgba(255,200,0,0.9)';ctx.lineWidth=1.5;
    ctx.strokeRect(bx-11,by+bobY-16,22,36);
  }
}

  const g=ctx.createLinearGradient(0,0,BASE_W,0);
  g.addColorStop(0,'#030B04');g.addColorStop(1,'#071508');
  ctx.fillStyle=g;ctx.fillRect(0,0,BASE_W,CH);

  ctx.font='12px Courier New';ctx.textAlign='center';
  ctx.fillText('◄',BASE_W-7,d0+dH/2+5);

  // Заголовок базы
  ctx.fillStyle='#325A22';ctx.font='bold 9px Courier New';ctx.textAlign='center';

  hline(8,BASE_W-16,18,'#162E0E');

  ctx.fillStyle='#FFD700';ctx.font='bold 24px Courier New';ctx.fillText(pRes,BASE_W/2,58);
  if(pCarry>0){

  const strPct=Math.min(1,pBaseStr/500);
  ctx.fillStyle='#4A7830';ctx.font='8px Courier New';ctx.textAlign='center';
  ctx.fillText('СИЛА БАЗЫ',BASE_W/2,92);
  const sbx=10,sby=96,sbw=BASE_W-22,sbh=10;
  ctx.fillStyle='#0A1808';ctx.fillRect(sbx,sby,sbw,sbh);
  const sc=strPct>0.66?'#44FF44':strPct>0.33?'#FFCC00':'#FF4444';
  ctx.fillStyle=sc;ctx.fillRect(sbx,sby,sbw*strPct,sbh);
  ctx.strokeStyle='#162E0E';ctx.lineWidth=0.8;ctx.strokeRect(sbx,sby,sbw,sbh);
  ctx.fillStyle='#FFD700';ctx.font='bold 11px Courier New';ctx.textAlign='center';
  ctx.fillText(pBaseStr,BASE_W/2,120);

  hline(8,BASE_W-16,125,'#162E0E');

  // Бойцы в базе — живая анимация
  const soldierCount=gs===GS.BASE?dockedLen:P.length;
  ctx.fillStyle='#4A7830';ctx.font='8px Courier New';ctx.textAlign='center';
  ctx.fillText('ЛИЧНЫЙ СОСТАВ',BASE_W/2,138);
  ctx.fillStyle='#8AB870';ctx.font='bold 12px Courier New';
  ctx.fillText(`${soldierCount} бойцов`,BASE_W/2,153);

  if(gs===GS.BASE){
    // Рисуем анимированных бойцов прямо в базе
    const visCount=Math.min(soldierCount,3);
    const startX=(BASE_W-(visCount*46))/2+20;
    for(let i=0;i<visCount;i++){
      const s=baseSoldierAnim[i]||{bx:startX+i*46,by:195,phase:i*1.5,state:'idle'};
      drawBaseSoldier(startX+i*46,195,s.phase,i===0);
    }
    if(soldierCount>3){
      ctx.fillStyle='#3A5828';ctx.font='7px Courier New';ctx.textAlign='center';
      ctx.fillText(`+${soldierCount-3} в казармах`,BASE_W/2,228);
    }
  } else {
    // На поле — HP-бары
    const mhp=3+upg.lives;
    for(let i=0;i<Math.min(soldierCount,5);i++){

function hline(x1,x2,y,col){
  ctx.strokeStyle=col;ctx.lineWidth=1;
  ctx.beginPath();ctx.moveTo(x1,y);ctx.lineTo(x2,y);ctx.stroke();
}
function drawWall(wx,wy,ww,wh,c1,c2,cBorder){
  ctx.fillStyle=c1;ctx.fillRect(wx,wy,ww,wh);
  for(let row=0;row<ROWS;row++){
    if(DOOR.includes(row))continue;
    const off=row%2===0?0:Math.floor(ww*0.5);
    ctx.fillStyle=row%2===0?c2:c1;
    ctx.fillRect(wx+off,row*CELL+1,ww-1,CELL-2);
    ctx.strokeStyle=cBorder;ctx.lineWidth=0.5;
    ctx.strokeRect(wx+off,row*CELL+1,ww-1,CELL-2);
  }
}

function loop(ts){
  if(last<0){last=ts;requestAnimationFrame(loop);return;}
  const dt=Math.min(ts-last,100);last=ts;
  update(dt);render();
  requestAnimationFrame(loop);
}
